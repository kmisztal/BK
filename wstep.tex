\section{Wstęp}

\begin{frame}{Cel wykładu}
  \raggedright
  Ta prezentacja jest przygotowana jako modułowy materiał do całodniowego wykładu
  o nowoczesnych narzędziach wykorzystywanych w badaniach klinicznych. Układ został
  rozbity na osobne pliki, dzięki czemu łatwo rozwijać każdy blok tematyczny niezależnie.

  \vspace{0.35cm}

  \begin{itemize}
    \item każdy główny temat ma własny plik w katalogu `sections/`,
    \item plik `main.tex` scala całość do jednej prezentacji,
    \item szkielet można wygodnie rozbudowywać o slajdy, grafiki i przykłady praktyczne.
  \end{itemize}
\end{frame}

\begin{frame}{Proponowana logika dnia}
  \begin{enumerate}
    \item zbudowanie wspólnego języka pojęć i mapy systemów eClinical,
    \item omówienie kluczowych klas systemów wykorzystywanych operacyjnie,
    \item pokazanie kierunków rozwoju: integracji, automatyzacji i AI.
  \end{enumerate}
\end{frame}
