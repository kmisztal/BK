\section{CTMS}

% SLIDE ID: sec06-goal
\BlockGoalFrame{\SlideID{sec06-goal}}{
  \item pokazać CTMS jako system zarządzania operacyjnego,
  \item odróżnić dane operacyjne od danych klinicznych,
  \item wskazać miejsca integracji z innymi modułami.
}

% SLIDE ID: sec06-core
\begin{frame}{CTMS jako warstwa operacyjna}
  \SlideID{sec06-core}
  \begin{itemize}
    \item CTMS porządkuje działania operacyjne związane z prowadzeniem badania.
    \item Wspiera planowanie wizyt monitorujących, statusy ośrodków i działania zespołu.
    \item Często pełni rolę systemu spinającego dane operacyjne z innymi modułami.
  \end{itemize}
\end{frame}
