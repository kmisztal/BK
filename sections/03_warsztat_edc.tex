\section{Warsztat EDC}

% SLIDE ID: sec03-goal
\BlockGoalFrame{\SlideID{sec03-goal}}{
  \item przejść od protokołu do logicznego modelu wizyt,
  \item zobaczyć, jak myśleć o polach, walidacjach i query,
  \item przygotować uczestników do samodzielnego projektowania prostych struktur.
}

% SLIDE ID: sec03-workshop
\begin{frame}{Warsztat projektowania EDC}
  \SlideID{sec03-workshop}
  \begin{itemize}
    \item Ta część może później zawierać ćwiczenie krok po kroku.
    \item Warto pokazać przejście od protokołu do struktury wizyt i formularzy.
    \item Dobrym dodatkiem będzie analiza najczęstszych błędów projektowych.
  \end{itemize}
\end{frame}

% SLIDE ID: sec03-format
\begin{frame}{Proponowany format warsztatu}
  \SlideID{sec03-format}
  \begin{enumerate}
    \item definicja punktów końcowych,
    \item model wizyt i zdarzeń,
    \item projekt formularza,
    \item przykładowe reguły walidacyjne.
  \end{enumerate}
\end{frame}

% SLIDE ID: sec03-demo
\DemoFrame{Budowa prostego eCRF}{
  Pokażemy przejście od strony protokołu do listy wizyt, formularzy i podstawowych reguł walidacyjnych.
}
