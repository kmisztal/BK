\section{Wprowadzenie i mapa systemów eClinical}

% SLIDE ID: sec01-goal
\BlockGoalFrame{\SlideID{sec01-goal}}{
  \item uporządkować mapę systemów używanych w badaniu klinicznym,
  \item pokazać zależności między procesem badania a narzędziami,
  \item przygotować grunt pod dalsze moduły praktyczne.
}

% SLIDE ID: sec01-role
\begin{frame}{Rola systemów eClinical}
  \SlideID{sec01-role}
  \begin{itemize}
    \item Systemy eClinical wspierają planowanie, prowadzenie i nadzór nad badaniem klinicznym.
    \item Każdy moduł odpowiada za inny fragment procesu: dane, operacje, bezpieczeństwo lub dokumentację.
    \item W praktyce największą wartość daje nie pojedynczy system, ale spójny ekosystem narzędzi.
  \end{itemize}
\end{frame}

% SLIDE ID: sec01-map
\begin{frame}{Mapa architektury rozwiązania}
  \SlideID{sec01-map}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
    \begin{block}{Warstwy}
      \begin{itemize}
        \item źródła danych,
        \item systemy operacyjne,
        \item raportowanie i nadzór.
      \end{itemize}
    \end{block}

    \column{0.48\textwidth}
    \begin{block}{Przykładowe moduły}
      EDC, ePRO, IWRS/IVRS, CTMS, eTMF, safety database, warstwa integracyjna.
    \end{block}
  \end{columns}
\end{frame}
