\BlockGoalFrame{\SlideID{sec07-goal}}{
  \item uporządkować rolę eTMF w gotowości inspekcyjnej,
  \item pokazać znaczenie metadanych i odpowiedzialności za dokumenty,
  \item zrozumieć, że kompletność to proces, a nie jednorazowy upload.
}

% SLIDE ID: sec07-core
\begin{frame}{Elektroniczny Trial Master File}
  \SlideID{sec07-core}
  \begin{itemize}
    \item eTMF porządkuje dokumentację badania w ujęciu regulatorowym i operacyjnym.
    \item Ważne są: kompletność, wersjonowanie, metadane oraz gotowość do inspekcji.
    \item W tej części można później dodać model referencyjny artefaktów i odpowiedzialności.
  \end{itemize}
\end{frame}
