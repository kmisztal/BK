\section{eTMF}

\begin{frame}{Elektroniczny Trial Master File}
  \begin{itemize}
    \item eTMF porządkuje dokumentację badania w ujęciu regulatorowym i operacyjnym.
    \item Ważne są: kompletność, wersjonowanie, metadane oraz gotowość do inspekcji.
    \item W tej części można później dodać model referencyjny artefaktów i odpowiedzialności.
  \end{itemize}
\end{frame}
