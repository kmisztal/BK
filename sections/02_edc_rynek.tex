\section{Rynek EDC}

% SLIDE ID: sec02-market-goal
\BlockGoalFrame{\SlideID{sec02-market-goal}}{
  \item uporządkować rynek systemów \texttt{EDC} i zobaczyć, że podobne funkcje mogą być osadzone w bardzo różnych modelach biznesowych,
  \item zrozumieć, czym różnią się platformy enterprise, rozwiązania akademickie i systemy wdrażane szybciej lub lokalnie,
  \item wiedzieć, na jakie ekrany, logikę pracy i ograniczenia warto patrzeć przy porównywaniu dostawców.
}

% SLIDE ID: sec02-market-why
\begin{frame}{Po co znać rynek EDC}
  \SlideID{sec02-market-why}
  \begin{itemize}
    \item Ten sam protokół można wdrożyć w różnych systemach, ale koszt, tempo konfiguracji i ergonomia pracy będą inne.
    \item Dla sponsora i CRO ważne są skalowalność, walidacja, integracje i standaryzacja procesu.
    \item Dla ośrodka ważniejsze bywają prostota formularza, szybkość pracy i przejrzysta obsługa query.
    \item Dla Państwa kluczowe jest nie tyle zapamiętanie nazw firm, ile rozpoznanie, jaki typ platformy najlepiej wspiera dany model badania.
  \end{itemize}
\end{frame}

% SLIDE ID: sec02-market-segments
\begin{frame}{Segmenty rynku EDC}
  \SlideID{sec02-market-segments}
  \begin{itemize}
    \item \textbf{Platformy enterprise}: zwykle wybierane do większych badań wieloośrodkowych, z naciskiem na standaryzację, walidację i integracje z resztą ekosystemu.
    \item \textbf{Rozwiązania zwinne i mid-market}: atrakcyjne tam, gdzie ważny jest szybszy start, prostsza konfiguracja i mniejszy narzut wdrożeniowy.
    \item \textbf{Środowisko akademickie}: większe znaczenie mają dostępność, koszt, lokalna administracja i możliwość samodzielnego uruchomienia projektu.
    \item \textbf{Lokalni dostawcy}: mogą lepiej odpowiadać na specyfikę językową, organizacyjną i operacyjną danego rynku.
  \end{itemize}

  \vspace{0.25cm}

  \begin{QuestionBox}
    Czy w praktyce „najlepszy system” istnieje obiektywnie, czy raczej zależy od typu badania, budżetu i dojrzałości organizacji?
  \end{QuestionBox}
\end{frame}

\begin{frame}{Oracle: rozwiązania ważne historycznie}
  \SlideID{sec02-market-oracle}
  \begin{itemize}
    \item \textbf{Oracle Clinical / RDC Onsite / InForm} to rozwiązania, które przez lata kształtowały sposób myślenia o pracy z \texttt{EDC}.
    \item Ich interfejs bywa dziś postrzegany jako starszy, ale pojęcia takie jak \texttt{casebook}, review, discrepancy czy audit trail były tam bardzo dobrze widoczne operacyjnie.
    \item To dobry materiał dydaktyczny, bo pokazuje, że rdzeń procesu pozostaje podobny mimo zmian w wyglądzie interfejsów.
    \item Warto pokazać: ekran logowania, przegląd pacjentów, drzewo formularzy i ekran pracy z query.
  \end{itemize}

  \vspace{0.25cm}

  \begin{WarningBox}
    Historyczny interfejs nie oznacza słabego procesu. Czasem starszy system jest mniej wygodny, ale nadal bardzo jasno pokazuje logikę danych i odpowiedzialności.
  \end{WarningBox}
\end{frame}

% SLIDE ID: sec02-market-selection
\begin{frame}{Jak w praktyce wybiera się EDC}
  \SlideID{sec02-market-selection}
  \begin{itemize}
    \item Czy badanie jest globalne, wieloośrodkowe i obudowane wieloma procesami sponsora?
    \item Czy trzeba szybko uruchomić projekt, a najważniejsza jest prostota pracy ośrodka?
    \item Czy organizacja potrzebuje pełnego ekosystemu \texttt{CTMS / eTMF / randomization / safety}, czy głównie dobrego formularza i kontroli danych?
    \item Czy kluczowe są lokalne wsparcie, język, szybkość konfiguracji i koszt utrzymania?
    \item Czy użytkownicy końcowi będą zaawansowani, czy trzeba projektować interfejs pod osoby, które z systemem pracują okazjonalnie?
  \end{itemize}
\end{frame}

% SLIDE ID: sec02-market-trends
\begin{frame}{W jaką stronę idzie rynek}
  \SlideID{sec02-market-trends}
  \begin{itemize}
    \item \textbf{Platformizacja}: \texttt{EDC} coraz rzadziej jest samotnym systemem, częściej staje się częścią większego środowiska \texttt{eClinical}.
    \item \textbf{Integracje}: rośnie znaczenie \texttt{API}, połączeń z \texttt{ePRO}, \texttt{eSource}, safety i narzędziami analitycznymi.
    \item \textbf{UX}: użytkownik ośrodka oczekuje prostszego interfejsu i mniejszej liczby kroków.
    \item \textbf{Automatyzacja}: systemy coraz częściej wspierają query, cleaning i monitoring sygnałami opartymi na regułach lub \texttt{AI}.
    \item \textbf{Gotowość regulacyjna}: sam nowoczesny wygląd nie wystarcza; nadal trzeba zapewnić walidację, ślad zmian i kontrolę dostępu.
  \end{itemize}
\end{frame}

% SLIDE ID: sec02-market-discussion
\DiscussionFrame{Czy rynek EDC naprawdę bardzo się różni?}{
  Jeśli spojrzymy na rdzeń procesu, większość systemów oferuje podobne elementy: formularze, walidacje, query, role i audit trail. Różnice stają się wyraźne dopiero wtedy, gdy oceniamy ergonomię pracy, tempo wdrożenia, integracje, model wsparcia i dojrzałość całego ekosystemu.
}

% SLIDE ID: sec02-market-summary
\begin{frame}{Wniosek z przeglądu rynku}
  \SlideID{sec02-market-summary}
  \begin{itemize}
    \item Rynek \texttt{EDC} jest dojrzały, ale bardzo zróżnicowany pod względem modelu użycia i otoczenia organizacyjnego.
    \item Największe platformy wygrywają skalą i ekosystemem, a mniejsze lub lokalne rozwiązania często wygrywają prostotą i szybkością wdrożenia.
    \item Z perspektywy użytkownika końcowego warto porównywać nie tylko listę funkcji, ale też logikę pracy na ekranie.
    \item Dobra demonstracja systemu powinna pokazać nie marketing, ale realny przepływ: pacjent, wizyta, formularz, walidacja, query i ślad zmian.
  \end{itemize}
\end{frame}
