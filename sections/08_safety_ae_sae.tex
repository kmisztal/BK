\section{Safety / AE / SAE}

% SLIDE ID: sec08-goal
\BlockGoalFrame{\SlideID{sec08-goal}}{
  \item uporządkować podstawy procesu AE/SAE od strony systemowej,
  \item pokazać miejsca krytyczne czasowo,
  \item przygotować grunt do omawiania integracji i jakości przepływu safety.
}

% SLIDE ID: sec08-core
\begin{frame}{Bezpieczeństwo uczestników i przepływ zgłoszeń}
  \SlideID{sec08-core}
  \begin{itemize}
    \item Proces safety obejmuje wykrycie, ocenę, klasyfikację i raportowanie zdarzeń.
    \item AE i SAE wymagają ścisłego reżimu czasowego oraz jednoznacznych odpowiedzialności.
    \item W praktyce ważna jest integracja z EDC, medical review i raportowaniem.
  \end{itemize}
\end{frame}

}{
