\section{AI w badaniach klinicznych}

% SLIDE ID: sec09-goal
\BlockGoalFrame{\SlideID{sec09-goal}}{
  \item oddzielić realne zastosowania AI od marketingowych haseł,
  \item wskazać obszary o największym potencjale praktycznym,
  \item omówić granice automatyzacji i potrzebę nadzoru człowieka.
}

% SLIDE ID: sec09-value
\begin{frame}{Gdzie AI wnosi realną wartość}
  \SlideID{sec09-value}
  \begin{itemize}
    \item AI może wspierać jakość danych, automatyzację procesów i analizę sygnałów.
    \item Największy efekt pojawia się tam, gdzie istnieją duże wolumeny danych i powtarzalne operacje.
    \item Ta sekcja może później zawierać zarówno przykłady użyteczne, jak i ograniczenia regulacyjne.
  \end{itemize}
\end{frame}

% SLIDE ID: sec09-questions
\begin{frame}{Pytania do rozwinięcia}
  \SlideID{sec09-questions}
  \begin{itemize}
    \item które procesy są gotowe na automatyzację już dziś,
    \item gdzie potrzebny jest pełny nadzór człowieka,
    \item jak dokumentować i walidować rozwiązania AI.
  \end{itemize}
\end{frame}

% SLIDE ID: sec09-discussion
\DiscussionFrame{AI jako wsparcie czy decyzja?}{
  W których momentach procesu badania AI może rekomendować, a w których nie powinna podejmować decyzji bez człowieka?
}
