\section{EDC / eCRF}

% SLIDE ID: sec02-goal
\BlockGoalFrame{\SlideID{sec02-goal}}{
  \item zrozumieć, że EDC jest centralnym systemem zbierania danych,
  \item zobaczyć, jak eCRF porządkuje logikę badania i wpisów,
  \item połączyć strukturę systemu z jakością danych i dalszą analizą.
}

% SLIDE ID: sec02-edc-definition
\begin{frame}{Czym jest EDC}
  \SlideID{sec02-edc-definition}
  \begin{DefinitionBox}
    EDC to system informatyczny, w którym zbierane, kontrolowane i porządkowane są dane z badania klinicznego.
  \end{DefinitionBox}

  \vspace{0.15cm}

  \begin{itemize}
    \item przechowuje dane pacjentów i wizyt,
    \item kontroluje poprawność wpisów,
    \item wspiera monitoring, cleaning i zamknięcie bazy.
  \end{itemize}
\end{frame}

% SLIDE ID: sec02-ecrf-definition
\begin{frame}{Czym jest eCRF}
  \SlideID{sec02-ecrf-definition}
  \begin{DefinitionBox}
    eCRF to elektroniczny formularz, który zbiera konkretne dane wymagane przez protokół badania.
  \end{DefinitionBox}

  \vspace{0.35cm}

  \begin{itemize}
    \item jest elementem systemu EDC,
    \item porządkuje dane według wizyt, formularzy i pól,
    \item przekłada protokół na praktyczne wpisy użytkownika.
  \end{itemize}
\end{frame}

% SLIDE ID: sec02-edc-vs-ecrf
\begin{frame}{EDC vs eCRF}
  \SlideID{sec02-edc-vs-ecrf}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
    \begin{beamercolorbox}[wd=\linewidth,sep=0.8em,rounded=true]{block body}
      \textbf{\textcolor{UAFMTeal}{EDC}} \par\medskip
      \begin{itemize}
        \item cały system zbierania danych,
        \item użytkownicy, role i uprawnienia,
        \item walidacje, query, audit trail,
        \item baza danych i eksporty.
      \end{itemize}
    \end{beamercolorbox}

    \column{0.48\textwidth}
    \begin{beamercolorbox}[wd=\linewidth,sep=0.8em,rounded=true]{block body}
      \textbf{\textcolor{UAFMOrange}{eCRF}} \par\medskip
      \begin{itemize}
        \item formularz w systemie EDC,
        \item pola danych dla wizyty,
        \item logika protokołu,
        \item miejsce wpisu użytkownika.
      \end{itemize}
    \end{beamercolorbox}
  \end{columns}

  \vspace{0.35cm}

  \begin{ExampleBox}
    EDC to całe środowisko pracy, a eCRF to formularz działający wewnątrz tego środowiska.
  \end{ExampleBox}
\end{frame}

% SLIDE ID: sec02-paper-vs-electronic
\begin{frame}{Papierowy CRF vs eCRF}
  \SlideID{sec02-paper-vs-electronic}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
    \begin{beamercolorbox}[wd=\linewidth,sep=0.8em,rounded=true]{block body}
      \textbf{\textcolor{UAFMOrange}{Papier}} \par\medskip
      \begin{itemize}
        \item wpis po fakcie,
        \item brak natychmiastowej kontroli,
        \item większe ryzyko braków i błędów.
      \end{itemize}
    \end{beamercolorbox}

    \column{0.48\textwidth}
    \begin{beamercolorbox}[wd=\linewidth,sep=0.8em,rounded=true]{block body}
      \textbf{\textcolor{UAFMTeal}{Elektroniczny eCRF}} \par\medskip
      \begin{itemize}
        \item wpis bezpośrednio do systemu,
        \item walidacja w momencie wpisu,
        \item pełna ścieżka zmian i szybszy cleaning.
      \end{itemize}
    \end{beamercolorbox}
  \end{columns}
\end{frame}

% SLIDE ID: sec02-paper-crf-history
\begin{frame}{Zanim był eCRF}
  \SlideID{sec02-paper-crf-history}
  \begin{itemize}
    \item dane zbierano w papierowych CRF,
    \item formularz był projektowany pod protokół,
    \item wpisy trafiały później do bazy przez data entry,
    \item kontrola jakości następowała z opóźnieniem.
  \end{itemize}

  \vspace{0.3cm}

  \begin{QuestionBox}
    Co mogło pójść źle, jeśli błąd odkrywano kilka tygodni po wizycie pacjenta?
  \end{QuestionBox}
\end{frame}

% SLIDE ID: sec02-paper-copying
\begin{frame}{Jak radzono sobie z kopiami}
  \SlideID{sec02-paper-copying}
  \begin{itemize}
    \item stosowano wielowarstwowe formularze,
    \item kopie powstawały przez papier samokopiujący,
    \item komplet dokumentów trafiał do sponsora, CRO lub archiwum,
    \item ryzykiem były braki stron, nieczytelność i rozbieżne wersje.
  \end{itemize}

  \vspace{0.3cm}

  \begin{ExampleBox}
    NCR / carbonless copy paper pozwalał tworzyć kopie CRF bez ręcznego przepisywania.
  \end{ExampleBox}
\end{frame}

% SLIDE ID: sec02-paper-still-exists
\begin{frame}{Papierowy CRF nie zniknął całkowicie}
  \SlideID{sec02-paper-still-exists}
  \begin{itemize}
    \item Papier nie jest dziś głównym modelem badań rejestracyjnych, ale nadal pojawia się w wybranych projektach.
    \item Dotyczy to zwłaszcza badań akademickich, badań terenowych i modeli hybrydowych opartych na skanowaniu formularzy.
    \item Nadal istnieją firmy oferujące projektowanie, druk i obsługę papierowych lub hybrydowych formularzy \texttt{CRF}.
    \item W praktyce papier częściej bywa etapem przejściowym niż docelowym modelem zarządzania danymi.
  \end{itemize}

  \vspace{0.3cm}

  \begin{ExampleBox}
    Przykłady: eClinical / ePartner Consulting oraz GSB Clinical.
  \end{ExampleBox}
\end{frame}

% SLIDE ID: sec02-paper-vendors
\begin{frame}{Kto nadal pracuje z papierowym CRF}
  \SlideID{sec02-paper-vendors}
  \begin{itemize}
    \item \textbf{eClinical / ePartner Consulting}: publicznie pokazuje ofertę projektowania zarówno papierowych, jak i elektronicznych \texttt{CRF}, wraz z logiką formularzy i przygotowaniem układu pod protokół.
    \item \textbf{GSB Clinical}: komunikuje usługi związane z projektowaniem, drukiem i logistyką papierowych materiałów badaniowych, w tym formularzy i dokumentów operacyjnych.
    \item Tego typu oferty nie oznaczają powrotu do dominacji papieru, ale pokazują, że w realnych badaniach nadal istnieją procesy hybrydowe.
    \item Dla Państwa to ważny sygnał praktyczny: nawet nowoczesny ekosystem \texttt{eClinical} czasem musi stykać się z dokumentacją papierową.
  \end{itemize}
\end{frame}

% SLIDE ID: sec02-paper-data-flow
\begin{frame}{Papierowy przepływ danych}
  \SlideID{sec02-paper-data-flow}
  \begin{center}
    \Large
    Pacjent $\rightarrow$ papierowy CRF $\rightarrow$ monitoring $\rightarrow$ data entry $\rightarrow$ query $\rightarrow$ korekta
  \end{center}

  \vspace{0.45cm}

  \begin{itemize}
    \item dużo pracy było wykonywane po fakcie,
    \item jakość zależała od czytelności i kompletności papieru,
    \item query często wracało do ośrodka z dużym opóźnieniem.
  \end{itemize}
\end{frame}

% SLIDE ID: sec02-validations
\begin{frame}{Walidacje: pierwsza linia kontroli}
  \SlideID{sec02-validations}
  \begin{itemize}
    \item sprawdzają kompletność, zakresy i logikę danych,
    \item mogą działać natychmiast przy wpisie lub po zapisaniu formularza,
    \item nie zastępują myślenia użytkownika, ale wyłapują część błędów wcześnie.
  \end{itemize}

  \vspace{0.3cm}

  \begin{ExampleBox}
    Wiek 250 lat, brak daty wizyty albo wynik niemożliwy dla danego badania laboratoryjnego.
  \end{ExampleBox}
\end{frame}

% SLIDE ID: sec02-market-context
\begin{frame}{Kilka nazw, które warto znać}
  \SlideID{sec02-market-context}
  \begin{itemize}
    \item \textbf{Medidata Rave}: jeden z najlepiej rozpoznawalnych systemów \texttt{EDC}; firma podaje, że \texttt{Rave} uruchomiono w 1999 roku.
    \item \textbf{Oracle InForm / RDC Onsite}: ważne historycznie rozwiązania pokazujące, jak rozwijał się model pracy z \texttt{eCRF}, query i review.
    \item \textbf{OpenClinica}, \textbf{Castor}, \textbf{REDCap}: rozwiązania częściej spotykane w mniejszych badaniach, środowisku akademickim albo projektach wymagających większej prostoty.
    \item Sama lista dostawców nie jest najważniejsza; ważniejsze jest zrozumienie, że większość systemów realizuje podobny rdzeń procesu, ale różni się ergonomią, zakresem integracji i dojrzałością operacyjną.
  \end{itemize}
\end{frame}

% SLIDE ID: sec02-why-edc-won
\begin{frame}{Dlaczego EDC wygrało}
  \SlideID{sec02-why-edc-won}
  \begin{itemize}
    \item walidacje działają wcześniej niż w papierze,
    \item monitor widzi dane szybciej,
    \item query można obsłużyć bez obiegu papieru,
    \item audit trail pokazuje pełną historię zmian.
  \end{itemize}

  \vspace{0.3cm}

  \begin{QuestionBox}
    Czy EDC usuwa błędy, czy tylko przesuwa je bliżej momentu wpisu?
  \end{QuestionBox}
\end{frame}

% SLIDE ID: sec02-query-management
\begin{frame}{Query management}
  \SlideID{sec02-query-management}
  \begin{itemize}
    \item query to pytanie o niejasny, brakujący lub niespójny wpis,
    \item może być generowane automatycznie albo przez monitora / data managera,
    \item jego celem jest wyjaśnienie danych, a nie „karanie” użytkownika.
  \end{itemize}
\end{frame}

% SLIDE ID: sec02-audit-trail
\begin{frame}{Audit trail}
  \SlideID{sec02-audit-trail}
  \begin{itemize}
    \item system zapisuje kto, kiedy i co zmienił,
    \item widać wartość pierwotną i wartość po zmianie,
    \item to kluczowe dla zgodności i wiarygodności danych.
  \end{itemize}
\end{frame}

% SLIDE ID: sec02-electronic-records
\begin{frame}{Co zmienił zapis elektroniczny}
  \SlideID{sec02-electronic-records}
  \begin{itemize}
    \item dane stały się łatwiejsze do śledzenia,
    \item podpis i zmiana muszą być przypisane do użytkownika,
    \item system musi chronić integralność rekordu,
    \item zgodność zależy od technologii i procedur.
  \end{itemize}

  \vspace{0.3cm}

  \begin{ExampleBox}
    Sam fakt, że dane są elektroniczne, nie wystarcza. System musi być kontrolowany, walidowany i używany zgodnie z procedurami.
  \end{ExampleBox}
\end{frame}

% SLIDE ID: sec02-roles
\begin{frame}{Role użytkowników}
  \SlideID{sec02-roles}
  \begin{itemize}
    \item badacz i koordynator wpisują lub uzupełniają dane,
    \item monitor przegląda wpisy i zgłasza query,
    \item data manager nadzoruje jakość danych i cleaning,
    \item sponsor ma dostęp zależny od roli i modelu badania.
  \end{itemize}
\end{frame}

% SLIDE ID: sec02-quality
\begin{frame}{Dlaczego jakość danych zaczyna się w EDC}
  \SlideID{sec02-quality}
  \begin{itemize}
    \item większość problemów jakościowych zaczyna się na etapie wpisu,
    \item dobra struktura formularza zmniejsza liczbę późniejszych pytań,
    \item im lepszy projekt EDC, tym mniej ręcznych korekt dalej w procesie.
  \end{itemize}
\end{frame}

% SLIDE ID: sec02-user-errors
\begin{frame}{Najczęstsze błędy użytkowników}
  \SlideID{sec02-user-errors}
  \begin{itemize}

    \item pominięcie pola obowiązkowego,
    \item brak reakcji na query,
    \item poprawa danych bez zrozumienia źródła problemu.
  \end{itemize}
\end{frame}
