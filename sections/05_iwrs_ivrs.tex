\section{IWRS / IVRS}

\begin{frame}{Randomizacja i zarządzanie leczeniem badanym}
  \begin{itemize}
    \item IWRS/IVRS wspiera randomizację, przydział leczenia i logistykę dostaw.
    \item System musi być jednocześnie prosty dla ośrodka i rygorystyczny proceduralnie.
    \item W tej sekcji warto później pokazać zależności między protokołem a logiką randomizacji.
  \end{itemize}
\end{frame}
