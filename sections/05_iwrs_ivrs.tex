\section{IWRS / IVRS}

% SLIDE ID: sec05-goal
\BlockGoalFrame{\SlideID{sec05-goal}}{
  \item zrozumieć rolę randomizacji i zarządzania leczeniem badanym,
  \item pokazać zależność między protokołem a konfiguracją systemu,
  \item omówić najważniejsze ryzyka operacyjne.
}

% SLIDE ID: sec05-core
\begin{frame}{Randomizacja i zarządzanie leczeniem badanym}
  \SlideID{sec05-core}
  \begin{itemize}
    \item IWRS/IVRS wspiera randomizację, przydział leczenia i logistykę dostaw.
    \item System musi być jednocześnie prosty dla ośrodka i rygorystyczny proceduralnie.
    \item W tej sekcji warto później pokazać zależności między protokołem a logiką randomizacji.
  \end{itemize}
\end{frame}

% SLIDE ID: sec05-warning
\begin{frame}{Ryzyko konfiguracyjne}
  \SlideID{sec05-warning}
  \begin{WarningBox}
    Nawet dobrze opisana randomizacja w protokole może zostać źle odwzorowana w systemie, jeśli zespół nie przełoży jej na precyzyjne reguły operacyjne i scenariusze wyjątków.
  \end{WarningBox}
\end{frame}
